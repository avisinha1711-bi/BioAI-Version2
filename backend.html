BioAI Backend
Simple Node.js API for Cancer Prediction

Minimal Backend Implementation

This lightweight backend provides the essential API endpoints needed for the BioAI frontend. It's simple but fully functional.

// server.js - Minimal BioAI Backend const express = require('express'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); // Health check endpoint app.get('/api/status', (req, res) => { res.json({ status: 'OK', message: 'BioAI API is running' }); }); // Prediction endpoint app.post('/api/predict', (req, res) => { const { values } = req.body; if (!values || values.length !== 4) { return res.status(400).json({ error: 'Invalid input. Expected 4 frequency values.' }); } // Simple prediction algorithm const probability = predictCancerRisk(values); res.json({ probability: probability, risk: probability >= 0.42 ? 'high' : 'low', recommendation: probability >= 0.42 ? 'MAY REQUIRE FURTHER DIAGNOSIS' : 'Regular monitoring recommended' }); }); // Simple prediction function function predictCancerRisk(values) { const NORMAL_MEANS = [1.2, 1.8, 2.8, 2.2]; const NORMAL_STDS = [0.4, 0.6, 0.7, 0.9]; let score = 0; for (let i = 0; i < values.length; i++) { const deviation = (values[i] - NORMAL_MEANS[i]) / NORMAL_STDS[i]; score += Math.max(0, deviation); } const rawProb = 1 / (1 + Math.exp(-score * 0.5 + 2)); return Math.min(0.99, rawProb + (Math.random() * 0.1 - 0.05)); } // Start server const PORT = process.env.PORT || 5000; app.listen(PORT, () => { console.log(`BioAI server running on port ${PORT}`); });

To set up this backend:

  1. Create a new directory for your backend
  2. Run npm init -y to create a package.json file
  3. Install dependencies: npm install express cors
  4. Save the above code as server.js
  5. Start the server: node server.js

API Endpoints

GET
/api/status
Health check endpoint to verify the API is running.
{ "status": "OK", "message": "BioAI API is running" }
POST
/api/predict
Accepts infrared frequency readings and returns a cancer risk prediction.

Request Body:

{ "values": [1.4, 1.9, 2.5, 2.8] }

Response:

{ "probability": 0.37, "risk": "low", "recommendation": "Regular monitoring recommended" }
